public class CartuchoDeTinta {

    private String codigo;
    private String cor;
    private double nivelAtualMl;
    private double capacidadeMl;

    public CartuchoDeTinta(String codigo, String cor, double capacidadeMl) {
        this.codigo = codigo;
        this.cor = cor;
        this.capacidadeMl = capacidadeMl;
        this.nivelAtualMl = capacidadeMl;
    }


    public void usarOuReabastecer(double ml) {
        if (ml > 0) {
            nivelAtualMl = Math.max(nivelAtualMl - ml, 0);
            System.out.println("Consumidos " + ml + "ml. Nível atual: " + nivelAtualMl + "ml");
        } else if (ml < 0) {
            double Reabastecido = -ml;
            nivelAtualMl = Math.min(nivelAtualMl + Reabastecido, capacidadeMl);
            System.out.println("Reabastecidos " + Reabastecido + " ml. Nível atual: " + nivelAtualMl + " ml");
        }
    }


    public void exibirInformacoes() {
        double percentual = (nivelAtualMl / capacidadeMl) * 100;
        System.out.println("Código: " + codigo);
        System.out.println("Cor: " + cor);
        System.out.println("Nível atual: " + nivelAtualMl + " ml (" + String.format("%.2f", percentual) + "%)");
    }
}
